    <script>
        // Simulate loading mechanics
        document.addEventListener('DOMContentLoaded', function() {
            // In a real app, this would be an API call
            setTimeout(function() {
                document.querySelector('.loading-spinner').classList.add('d-none');
                document.getElementById('mechanicsList').classList.remove('d-none');
            }, 1500);
            
            // Set user location from previous page or geolocation
            if(localStorage.getItem('userLocation')) {
                document.getElementById('userLocation').textContent = localStorage.getItem('userLocation');
            }
        });
        
        // Handle mechanic selection
        function selectMechanic(mechanicId) {
            // Remove selected class from all cards
            document.querySelectorAll('.mechanic-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Add selected class to clicked card
            const selectedCard = document.getElementById(mechanicId);
            selectedCard.classList.add('selected');
            
            // Update all buttons
            document.querySelectorAll('.btn-select').forEach(btn => {
                btn.innerHTML = 'Select This Mechanic';
                btn.classList.remove('btn-primary');
                btn.classList.add('btn-outline-primary');
            });
            
            // Update selected button
            const selectedBtn = selectedCard.querySelector('.btn-select');
            selectedBtn.innerHTML = '<i class="bi bi-check-circle me-2"></i> Selected';
            selectedBtn.classList.remove('btn-outline-primary');
            selectedBtn.classList.add('btn-primary');
            
            // Update confirmation section
            const mechanicName = selectedCard.querySelector('h5').textContent;
            document.getElementById('selectedMechanicName').textContent = mechanicName;
            document.getElementById('confirmationSection').classList.remove('d-none');
            
            // Scroll to confirmation section
            document.getElementById('confirmationSection').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }
        
        // Handle confirmation
        function confirmSelection() {
            // In a real app, this would send the selection to your backend
            alert('Mechanic confirmed! You will receive an OTP for verification shortly.');
            
            // Redirect to tracking page
            // window.location.href = 'tracking.html';
        }
